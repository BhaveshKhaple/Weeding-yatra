**File:** `src/lib/lenis.ts`
```typescript
import Lenis from '@studio-freight/lenis'
import { gsap } from './gsap'

let lenisInstance: Lenis | null = null

export function createLenis(): Lenis {
  if (lenisInstance) return lenisInstance

  lenisInstance = new Lenis({
    duration: 1.2,
    easing: (t) => Math.min(1, 1.001 - Math.pow(2, -10 * t)),
    smoothWheel: true,
  })

  // Sync Lenis RAF with GSAP ticker for ScrollTrigger compatibility
  gsap.ticker.add((time) => {
    lenisInstance?.raf(time * 1000)
  })
  gsap.ticker.lagSmoothing(0)

  return lenisInstance
}

export function destroyLenis(): void {
  lenisInstance?.destroy()
  lenisInstance = null
}
```

```typescript
/**
 * Generate a URL-safe wedding slug from couple names.
 * Example: "Priya" + "Rahul" → "priya-rahul-k8j2"
 */
export function generateSlug(brideName: string, groomName: string): string {
  const base = `${brideName}-${groomName}`
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, '-')
    .replace(/^-|-$/g, '')
  const suffix = Math.random().toString(36).substring(2, 6)
  return `${base}-${suffix}`
}
```
